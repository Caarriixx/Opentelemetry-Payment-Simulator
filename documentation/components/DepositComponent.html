import { Component, OnInit } from '@angular/core';
import {Router} from "@angular/router";
import {FormBuilder, FormGroup, ReactiveFormsModule, Validators} from "@angular/forms";
import { AccountService } from '../services/account.service';
import { User } from '../models/account.model';
import {CommonModule} from "@angular/common";
import { MaterialModule } from '../app/material/material.module';

@Component({
  selector: 'app-deposit',
  templateUrl: './deposit.component.html',
  standalone: true,
  imports: [
    ReactiveFormsModule,
    CommonModule,
    MaterialModule
],
  styleUrls: ['./deposit.component.css']
})
export class DepositComponent implements OnInit {
  account!: User;
  tries: number = 3;
  depositForm: FormGroup = this.formBuilder.group({}); // Inicialización en la declaración

  constructor(private accountService: AccountService, private formBuilder: FormBuilder, private router: Router) {
    // Modificamos la validación de la contraseña para que requiera al menos 6 caracteres
    this.depositForm = this.formBuilder.group({
      amount: ['', [Validators.required, Validators.min(0.01), this.maxTwoDecimalValidator]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }

  // Custom validator for max two decimal places
  maxTwoDecimalValidator(control: any): { [key: string]: boolean } | null {
    const value = control.value;
    if (value && !/^(\d+(\.\d{1,2})?)$/.test(value)) {
      return { maxTwoDecimals: true };
    }
    return null;
  }

  onSubmit() {
    console.log("submit hecho");
    if (this.depositForm.invalid) {
      alert('Please fill in all the required fields.');
      return;
    }

    if(this.tries != 0) {
      if (this.depositForm.get('password')?.value != this.account.password) {
        this.tries -= 1;
        alert(`Passwords do not match. You have ${this.tries} tries left.`);
        if (this.tries == 0) {
          alert('You have surpassed the maximum amount of tries. Please log again.');
          this.router.navigate(['/login']);
          return;
        }
        return;
      }
    }

    const amount = parseFloat(this.depositForm.get('amount')?.value);
    if (isNaN(amount) || amount <= 0) {
      alert('Please enter a valid amount.');
      return;
    }
    /*
    alert('Deposit confirmed!');
    this.account.balance += amount;
    */
    setTimeout(() => {
      this.router.navigate(['/account']);
    }, 1000);
  }

  goBack() {
    console.log("vuelve account");
    this.router.navigate(['/account']);
  }

  ngOnInit(): void {
    this.account = this.accountService.getAccount();
  }
}
